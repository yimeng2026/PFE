## 2. 部署架构

### 2.1 单机部署方案

```yaml
# docker-compose.single.yml
version: '3.8'

services:
  # 主应用服务
  aws:
    image: agent-writing-system:latest
    container_name: aws-main
    ports:
      - "8000:8000"  # API端口
      - "9090:9090"  # 监控端口
    volumes:
      - ./config.yaml:/app/config.yaml:ro
      - ./outputs:/app/outputs
      - ./logs:/app/logs
      - ./.env:/app/.env:ro
    environment:
      - AWS_CONFIG_PATH=/app/config.yaml
      - LOG_LEVEL=INFO
    command: >
      python -m aws.server
      --config /app/config.yaml
      --host 0.0.0.0
      --port 8000
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  # 可选：Redis消息总线
  redis:
    image: redis:7-alpine
    container_name: aws-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 3

  # 可选：Prometheus监控
  prometheus:
    image: prom/prometheus:latest
    container_name: aws-prometheus
    ports:
      - "9091:9090"
    volumes:
      - ./monitoring/prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus_data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
    restart: unless-stopped

  # 可选：Grafana仪表盘
  grafana:
    image: grafana/grafana:latest
    container_name: aws-grafana
    ports:
      - "3000:3000"
    volumes:
      - grafana_data:/var/lib/grafana
      - ./monitoring/grafana/dashboards:/etc/grafana/provisioning/dashboards:ro
      - ./monitoring/grafana/datasources:/etc/grafana/provisioning/datasources:ro
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin
    restart: unless-stopped

volumes:
  redis_data:
  prometheus_data:
  grafana_data:
```

### 2.2 分布式部署方案

```yaml
# docker-compose.distributed.yml
version: '3.8'

services:
  # API网关
  api-gateway:
    image: agent-writing-system:gateway
    ports:
      - "80:80"
      - "443:443"
    environment:
      - UPSTREAM_SERVERS=aws-api-1:8000,aws-api-2:8000
    depends_on:
      - aws-api-1
      - aws-api-2
    restart: unless-stopped

  # API服务实例1
  aws-api-1:
    image: agent-writing-system:latest
    environment:
      - NODE_ID=api-1
      - ROLE=api
      - COORDINATOR_URL=http://coordinator:8000
    configs:
      - source: aws_config
        target: /app/config.yaml
    secrets:
      - aws_env
    restart: unless-stopped
    deploy:
      resources:
        limits:
          cpus: '2'
          memory: 4G
        reservations:
          cpus: '1'
          memory: 2G

  # API服务实例2
  aws-api-2:
    image: agent-writing-system:latest
    environment:
      - NODE_ID=api-2
      - ROLE=api
      - COORDINATOR_URL=http://coordinator:8000
    configs:
      - source: aws_config
        target: /app/config.yaml
    secrets:
      - aws_env
    restart: unless-stopped
    deploy:
      resources:
        limits:
          cpus: '2'
          memory: 4G

  # 协调器服务
  coordinator:
    image: agent-writing-system:latest
    environment:
      - NODE_ID=coordinator
      - ROLE=coordinator
      - REDIS_URL=redis://redis:6379
    ports:
      - "8001:8000"
    depends_on:
      - redis
    configs:
      - source: aws_config
        target: /app/config.yaml
    secrets:
      - aws_env
    restart: unless-stopped
    deploy:
      resources:
        limits:
          cpus: '2'
          memory: 4G

  # Agent工作节点池
  agent-worker:
    image: agent-writing-system:latest
    environment:
      - ROLE=worker
      - COORDINATOR_URL=http://coordinator:8000
    configs:
      - source: aws_config
        target: /app/config.yaml
    secrets:
      - aws_env
    deploy:
      replicas: 5
      resources:
        limits:
          cpus: '2'
          memory: 4G
    restart: unless-stopped

  # Redis集群
  redis:
    image: redis:7-alpine
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    deploy:
      replicas: 3
    restart: unless-stopped

  # 对象存储（MinIO）
  minio:
    image: minio/minio:latest
    command: server /data --console-address ":9001"
    ports:
      - "9000:9000"
      - "9001:9001"
    volumes:
      - minio_data:/data
    environment:
      - MINIO_ROOT_USER=admin
      - MINIO_ROOT_PASSWORD=${MINIO_PASSWORD}
    restart: unless-stopped

configs:
  aws_config:
    file: ./config.yaml

secrets:
  aws_env:
    file: ./.env

volumes:
  redis_data:
  minio_data:
```

class ResourceScheduler:
    """
    资源调度器
    管理Agent集群的资源分配和负载均衡
    """
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.agent_pool: Dict[str, Dict] = {}  # agent_id -> resource_info
        self.task_queue: asyncio.Queue = asyncio.Queue()
        self.running = False
    
    async def register_agent(self, agent_id: str, capabilities: Dict):
        """注册Agent到资源池"""
        self.agent_pool[agent_id] = {
            "capabilities": capabilities,
            "status": "idle",
            "current_load": 0,
            "max_load": capabilities.get("max_concurrent_tasks", 3),
            "last_heartbeat": time.time()
        }
    
    async def schedule_task(self, task: Dict[str, Any]) -> Optional[str]:
        """
        调度任务到合适的Agent
        
        策略：
        1. 能力匹配（Agent角色与任务类型匹配）
        2. 负载均衡（选择负载最轻的Agent）
        3. 亲和性（尽量将相关任务分配给同一Agent）
        """
        task_type = task.get("type")
        task_requirements = task.get("requirements", {})
        
        # 筛选符合能力要求的Agent
        eligible_agents = []
        for agent_id, info in self.agent_pool.items():
            if info["status"] != "healthy":
                continue
            if info["current_load"] >= info["max_load"]:
                continue
            if not self._check_capability_match(info["capabilities"], task_type):
                continue
            eligible_agents.append((agent_id, info))
        
        if not eligible_agents:
            return None
        
        # 负载均衡：选择当前负载最轻的Agent
        selected_agent = min(eligible_agents, key=lambda x: x[1]["current_load"])
        
        # 增加Agent负载计数
        self.agent_pool[selected_agent[0]]["current_load"] += 1
        
        return selected_agent[0]
    
    def _check_capability_match(self, capabilities: Dict, task_type: str) -> bool:
        """检查Agent能力是否匹配任务类型"""
        supported_types = capabilities.get("supported_task_types", [])
        return task_type in supported_types
    
    async def release_agent(self, agent_id: str):
        """释放Agent（任务完成）"""
        if agent_id in self.agent_pool:
            self.agent_pool[agent_id]["current_load"] = max(
                0, self.agent_pool[agent_id]["current_load"] - 1
            )
    
    async def auto_scale(self):

        自动扩缩容
        根据负载动态调整Agent数量

        total_capacity = sum(info["max_load"] for info in self.agent_pool.values())

import logging

from pythonjsonlogger import jsonlogger

class CustomJsonFormatter(jsonlogger.JsonFormatter):
    """自定义JSON日志格式"""
    
    def add_fields(self, log_record, record, message_dict):
        super(CustomJsonFormatter, self).add_fields(log_record, record, message_dict)
        log_record['timestamp'] = datetime.utcnow().isoformat()
        log_record['level'] = record.levelname
        log_record['source'] = f"{record.name}:{record.lineno}"

def setup_logging(log_level: str = "INFO", log_file: Optional[str] = None):
    """配置日志"""
    log_handler = logging.StreamHandler()
    formatter = CustomJsonFormatter('%(timestamp)s %(level)s %(name)s %(message)s')
    log_handler.setFormatter(formatter)
    
    root_logger = logging.getLogger()
    root_logger.setLevel(getattr(logging, log_level))
    root_logger.addHandler(log_handler)
    
    if log_file:
        file_handler = logging.handlers.RotatingFileHandler(
            log_file,
            maxBytes=10485760,  # 10MB
            backupCount=10
        )
        file_handler.setFormatter(formatter)
        root_logger.addHandler(file_handler)

        for rule in self.ALERT_RULES:
            if rule["condition"](system_stats):

            "timestamp": datetime.now().isoformat(),
            "stats": stats
        }
        
        for channel in rule["channels"]:
            if channel == "email":
                await self._send_email_alert(alert)
            elif channel == "slack":
