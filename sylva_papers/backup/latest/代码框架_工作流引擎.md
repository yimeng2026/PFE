# Agent集群写稿系统 - 工作流引擎与协调器

## 1. 工作流引擎 (WorkflowEngine)

```python
import asyncio
from typing import Dict, List, Optional, Callable, Any
from dataclasses import dataclass, field
from datetime import datetime
import logging

class WorkflowEngine:
    """
    工作流引擎 - 核心调度器
    负责管理整个写稿流程的状态转换和任务调度
    """
    
    def __init__(self, coordinator: 'Coordinator', config: Dict[str, Any]):
        self.coordinator = coordinator
        self.config = config
        self.logger = logging.getLogger(__name__)
        
        # 阶段处理器注册表
        self.phase_handlers: Dict[WorkflowPhase, Callable] = {}
        self._register_default_handlers()
        
        # 状态转换规则
        self.transition_rules: Dict[WorkflowPhase, List[WorkflowPhase]] = {
            WorkflowPhase.IDLE: [WorkflowPhase.MULTI_WRITING],
            WorkflowPhase.MULTI_WRITING: [WorkflowPhase.DRAFT_SELECTION],
            WorkflowPhase.DRAFT_SELECTION: [WorkflowPhase.DEEP_EDITING, WorkflowPhase.MULTI_WRITING],
            WorkflowPhase.DEEP_EDITING: [WorkflowPhase.EXPERT_REVIEW, WorkflowPhase.MULTI_WRITING],
            WorkflowPhase.EXPERT_REVIEW: [WorkflowPhase.PEER_REVIEW, WorkflowPhase.REVISION, WorkflowPhase.REJECTED],
            WorkflowPhase.PEER_REVIEW: [WorkflowPhase.REVISION, WorkflowPhase.FINAL_DECISION, WorkflowPhase.REJECTED],
            WorkflowPhase.REVISION: [WorkflowPhase.PEER_REVIEW, WorkflowPhase.FINAL_DECISION],
            WorkflowPhase.FINAL_DECISION: [WorkflowPhase.PUBLISHED, WorkflowPhase.REJECTED, WorkflowPhase.REVISION],
        }
    
    def _register_default_handlers(self):
        """注册默认的阶段处理器"""
        self.phase_handlers[WorkflowPhase.MULTI_WRITING] = self._handle_multi_writing
        self.phase_handlers[WorkflowPhase.DRAFT_SELECTION] = self._handle_draft_selection
        self.phase_handlers[WorkflowPhase.DEEP_EDITING] = self._handle_deep_editing
        self.phase_handlers[WorkflowPhase.EXPERT_REVIEW] = self._handle_expert_review
        self.phase_handlers[WorkflowPhase.PEER_REVIEW] = self._handle_peer_review
        self.phase_handlers[WorkflowPhase.REVISION] = self._handle_revision
        self.phase_handlers[WorkflowPhase.FINAL_DECISION] = self._handle_final_decision
    
    async def execute_workflow(self, workflow_id: str, topic: str, requirements: Dict[str, Any]) -> Dict[str, Any]:
        """
        执行完整工作流
        
        Args:
            workflow_id: 工作流唯一标识
            topic: 论文主题
            requirements: 写作要求
            
        Returns:
            工作流执行结果
        """
        # 创建工作流状态
        workflow_state = WorkflowState(
            workflow_id=workflow_id,
            topic=topic,
            config=self.config,
            data_store={"requirements": requirements, "topic": topic}
        )
        
        # 注册到Coordinator
        await self.coordinator.register_workflow(workflow_state)
        
        try:
            # 开始执行流程
            await self._advance_phase(workflow_state, WorkflowPhase.MULTI_WRITING)
            
            # 主循环
            while workflow_state.status == WorkflowStatus.RUNNING:
                current_phase = workflow_state.current_phase
                
                if current_phase == WorkflowPhase.PUBLISHED:
                    workflow_state.status = WorkflowStatus.COMPLETED
                    break
                    
                if current_phase == WorkflowPhase.REJECTED:
                    workflow_state.status = WorkflowStatus.FAILED
                    break
                
                # 执行当前阶段
                handler = self.phase_handlers.get(current_phase)
                if handler:
                    phase_result = await handler(workflow_state)
                    
                    # 检查阶段结果
                    if phase_result.status == "failed":
                        # 失败处理：尝试重试或终止
                        if await self._handle_phase_failure(workflow_state, phase_result):
                            continue
                        else:
                            workflow_state.status = WorkflowStatus.FAILED
                            break
                    
                    # 记录阶段结果
                    workflow_state.transition_to(current_phase, phase_result)
                    
                    # 确定下一阶段
                    next_phase = await self._determine_next_phase(workflow_state, phase_result)
                    if next_phase:
                        await self._advance_phase(workflow_state, next_phase)
                else:
                    self.logger.error(f"No handler for phase {current_phase}")
                    workflow_state.status = WorkflowStatus.FAILED
                    break
                
                # 检查超时
                if await self._check_timeout(workflow_state):
                    workflow_state.status = WorkflowStatus.TIMEOUT
                    break
            
            # 生成最终结果
            return self._generate_final_result(workflow_state)
            
        except Exception as e:
            self.logger.exception("Workflow execution failed")
            workflow_state.status = WorkflowStatus.FAILED
            return {
                "workflow_id": workflow_id,
                "status": "failed",
                "error": str(e),
                "partial_result": workflow_state.to_summary_dict()
            }
    
    async def _handle_multi_writing(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 1: 多写并行
        启动多个写作Agent并行创作
        """
        self.logger.info(f"[{state.workflow_id}] Starting multi-writing phase")
        
        requirements = state.get_data("requirements", {})
        topic = state.topic
        writer_count = self.config.get("writer_count", 5)
        
        # 创建写作任务
        writing_tasks = []
        for i in range(writer_count):
            task_id = f"write_{i}_{state.workflow_id}"
            style = self.config.get("writer_styles", ["balanced"])[i % 5]
            
            task = {
                "task_id": task_id,
                "topic": topic,
                "requirements": requirements,
                "writing_style": style,
                "agent_role": "writer"
            }
            writing_tasks.append(task)
        
        # 并行执行写作任务
        start_time = datetime.now()
        
        try:
            # 通过Coordinator调度Agent执行
            draft_futures = [
                self.coordinator.dispatch_task("writer", task)
                for task in writing_tasks
            ]
            
            # 等待所有写作完成（带超时）
            timeout = self.config.get("writing_timeout", 1800)
            drafts = await asyncio.wait_for(
                asyncio.gather(*draft_futures, return_exceptions=True),
                timeout=timeout
            )
            
            # 处理结果
            successful_drafts = []
            failed_count = 0
            
            for i, result in enumerate(drafts):
                if isinstance(result, Exception):
                    self.logger.warning(f"Writing task {i} failed: {result}")
                    failed_count += 1
                else:
                    successful_drafts.append(result)
            
            # 保存草稿
            state.set_data("drafts", successful_drafts)
            state.set_data("writing_stats", {
                "total_requested": writer_count,
                "successful": len(successful_drafts),
                "failed": failed_count
            })
            
            duration = (datetime.now() - start_time).total_seconds()
            
            return PhaseResult(
                phase=WorkflowPhase.MULTI_WRITING,
                status="success" if len(successful_drafts) > 0 else "partial_success",
                output_data={"drafts": successful_drafts},
                duration_seconds=duration
            )
            
        except asyncio.TimeoutError:
            return PhaseResult(
                phase=WorkflowPhase.MULTI_WRITING,
                status="failed",
                error_message="Writing phase timeout",
                duration_seconds=(datetime.now() - start_time).total_seconds()
            )
    
    async def _handle_draft_selection(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 2: 初稿筛选
        从多版本中选择最优草稿
        """
        self.logger.info(f"[{state.workflow_id}] Starting draft selection")
        
        drafts = state.get_data("drafts", [])
        if not drafts:
            return PhaseResult(
                phase=WorkflowPhase.DRAFT_SELECTION,
                status="failed",
                error_message="No drafts available for selection"
            )
        
        # 如果有多个草稿，使用质量评估选择最优
        if len(drafts) == 1:
            selected = drafts[0]
        else:
            # 评估所有草稿
            scored_drafts = []
            for draft in drafts:
                score = draft.get("estimated_quality", 50)
                scored_drafts.append((score, draft))
            
            # 排序选择最高分
            scored_drafts.sort(reverse=True, key=lambda x: x[0])
            selected = scored_drafts[0][1]
            
            # 记录选择理由
            state.set_data("selection_reason", {
                "selected_draft_id": selected.get("draft_id"),
                "selected_score": scored_drafts[0][0],
                "all_scores": [(d.get("draft_id"), s) for s, d in scored_drafts]
            })
        
        state.set_data("selected_draft", selected)
        state.set_data("current_draft", selected)
        
        return PhaseResult(
            phase=WorkflowPhase.DRAFT_SELECTION,
            status="success",
            output_data={"selected_draft": selected}
        )
    
    async def _handle_deep_editing(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 3: 深度编辑
        对选中草稿进行语言润色和结构优化
        """
        self.logger.info(f"[{state.workflow_id}] Starting deep editing")
        
        draft = state.get_data("current_draft")
        if not draft:
            return PhaseResult(
                phase=WorkflowPhase.DEEP_EDITING,
                status="failed",
                error_message="No draft available for editing"
            )
        
        # 调度编辑Agent
        edit_task = {
            "draft": draft,
            "style_guide": self.config.get("style_guide", "academic"),
            "quality_threshold": self.config.get("quality_threshold", 70)
        }
        
        try:
            edit_result = await self.coordinator.dispatch_task("editor", edit_task)
            
            # 更新当前草稿
            edited_draft = edit_result.get("edited_version")
            state.set_data("current_draft", edited_draft)
            state.set_data("edit_report", edit_result.get("edit_report"))
            
            return PhaseResult(
                phase=WorkflowPhase.DEEP_EDITING,
                status="success",
                output_data=edit_result
            )
            
        except Exception as e:
            return PhaseResult(
                phase=WorkflowPhase.DEEP_EDITING,
                status="failed",
                error_message=str(e)
            )
    
    async def _handle_expert_review(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 4: 专家评审
        领域专家进行技术审核
        """
        self.logger.info(f"[{state.workflow_id}] Starting expert review")
        
        draft = state.get_data("current_draft")
        
        review_task = {
            "manuscript": draft,
            "domain": state.get_data("requirements", {}).get("domain", "general"),
            "review_type": "expert"
        }
        
        try:
            expert_report = await self.coordinator.dispatch_task("expert", review_task)
            
            state.set_data("expert_report", expert_report)
            
            # 根据专家意见决定流向
            verdict = expert_report.get("verdict", "major_revision")
            
            return PhaseResult(
                phase=WorkflowPhase.EXPERT_REVIEW,
                status="success",
                output_data={
                    "expert_report": expert_report,
                    "recommendation": verdict
                }
            )
            
        except Exception as e:
            return PhaseResult(
                phase=WorkflowPhase.EXPERT_REVIEW,
                status="failed",
                error_message=str(e)
            )
    
    async def _handle_peer_review(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 5: 同行审稿
        模拟多审稿人审稿流程
        """
        self.logger.info(f"[{state.workflow_id}] Starting peer review")
        
        draft = state.get_data("current_draft")
        reviewer_count = self.config.get("reviewer_count", 3)
        
        # 创建多个审稿任务
        review_tasks = []
        for i in range(reviewer_count):
            review_tasks.append({
                "manuscript": draft,
                "reviewer_style": f"reviewer_{i}",
                "review_type": "peer"
            })
        
        try:
            # 并行审稿
            review_futures = [
                self.coordinator.dispatch_task("reviewer", task)
                for task in review_tasks
            ]
            
            timeout = self.config.get("review_timeout", 1800)
            review_reports = await asyncio.wait_for(
                asyncio.gather(*review_futures, return_exceptions=True),
                timeout=timeout
            )
            
            # 过滤有效报告
            valid_reports = [
                r for r in review_reports 
                if not isinstance(r, Exception)
            ]
            
            state.set_data("review_reports", valid_reports)
            
            # 计算共识
            verdicts = [r.get("verdict", "major_revision") for r in valid_reports]
            consensus = self._calculate_consensus(verdicts)
            
            return PhaseResult(
                phase=WorkflowPhase.PEER_REVIEW,
                status="success",
                output_data={
                    "review_reports": valid_reports,
                    "consensus": consensus,
                    "verdicts_distribution": {v: verdicts.count(v) for v in set(verdicts)}
                }
            )
            
        except asyncio.TimeoutError:
            return PhaseResult(
                phase=WorkflowPhase.PEER_REVIEW,
                status="failed",
                error_message="Peer review phase timeout"
            )
    
    async def _handle_revision(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 6: 修改阶段
        根据审稿意见修改论文
        """
        self.logger.info(f"[{state.workflow_id}] Starting revision phase")
        
        current_draft = state.get_data("current_draft")
        review_reports = state.get_data("review_reports", [])
        expert_report = state.get_data("expert_report")
        
        # 整合所有评论
        all_comments = []
        for report in review_reports:
            all_comments.extend(report.get("comments", []))
        
        if expert_report:
            all_comments.extend(expert_report.get("comments", []))
        
        revision_task = {
            "draft": current_draft,
            "comments": all_comments,
            "revision_count": len(state.phase_history)  # 用于追踪修改轮数
        }
        
        try:
            # 调度写作Agent进行修改（可能需要原作者或专门修改Agent）
            revised_draft = await self.coordinator.dispatch_task("writer", revision_task)
            
            # 更新草稿版本
            revised_draft["version"] = current_draft.get("version", 1) + 1
            revised_draft["parent_draft_id"] = current_draft.get("draft_id")
            
            state.set_data("current_draft", revised_draft)
            
            # 记录修改历史
            revision_history = state.get_data("revision_history", [])
            revision_history.append({
                "version": revised_draft["version"],
                "comments_addressed": len(all_comments),
                "timestamp": datetime.now().isoformat()
            })
            state.set_data("revision_history", revision_history)
            
            return PhaseResult(
                phase=WorkflowPhase.REVISION,
                status="success",
                output_data={"revised_draft": revised_draft}
            )
            
        except Exception as e:
            return PhaseResult(
                phase=WorkflowPhase.REVISION,
                status="failed",
                error_message=str(e)
            )
    
    async def _handle_final_decision(self, state: WorkflowState) -> PhaseResult:
        """
        Phase 7: 最终决策
        主编Agent做出最终裁决
        """
        self.logger.info(f"[{state.workflow_id}] Starting final decision")
        
        current_draft = state.get_data("current_draft")
        review_reports = state.get_data("review_reports", [])
        revision_history = state.get_data("revision_history", [])
        
        decision_task = {
            "manuscript": current_draft,
            "review_reports": review_reports,
            "revision_history": revision_history
        }
        
        try:
            final_decision = await self.coordinator.dispatch_task("editor_in_chief", decision_task)
            
            verdict = final_decision.get("decision", "rejected")
            
            # 根据决策更新状态
            if verdict == "accepted":
                state.set_data("final_manuscript", final_decision.get("accepted_manuscript"))
            
            return PhaseResult(
                phase=WorkflowPhase.FINAL_DECISION,
                status="success",
                output_data=final_decision
            )
            
        except Exception as e:
            return PhaseResult(
                phase=WorkflowPhase.FINAL_DECISION,
                status="failed",
                error_message=str(e)
            )
    
    async def _determine_next_phase(
        self, 
        state: WorkflowState, 
        phase_result: PhaseResult
    ) -> Optional[WorkflowPhase]:
        """
        根据当前阶段结果确定下一阶段
        这是状态转换的核心逻辑
        """
        current = state.current_phase
        
        # 特殊处理规则
        if current == WorkflowPhase.EXPERT_REVIEW:
            recommendation = phase_result.output_data.get("recommendation", "major_revision")
            if recommendation == "reject":
                return WorkflowPhase.REJECTED
            elif recommendation == "accept":
                return WorkflowPhase.PEER_REVIEW  # 仍需同行审稿
        
        elif current == WorkflowPhase.PEER_REVIEW:
            consensus = phase_result.output_data.get("consensus", {})
            verdict = consensus.get("dominant_verdict", "major_revision")
            
            if verdict == "accept":
                return WorkflowPhase.FINAL_DECISION
            elif verdict in ["minor_revision", "major_revision"]:
                # 检查修改轮数
                revision_count = len([p for p in state.phase_history if p.phase == WorkflowPhase.REVISION])
                max_iterations = state.config.get("max_iterations", 3)
                
                if revision_count >= max_iterations:
                    return WorkflowPhase.FINAL_DECISION  # 强制进入终审
                else:
                    return WorkflowPhase.REVISION
            elif verdict == "reject":
                return WorkflowPhase.REJECTED
        
        elif current == WorkflowPhase.REVISION:
            # 修改后回到审稿
            return WorkflowPhase.PEER_REVIEW
        
        elif current == WorkflowPhase.FINAL_DECISION:
            decision = phase_result.output_data.get("decision", "rejected")
            if decision == "accepted":
                return WorkflowPhase.PUBLISHED
            elif decision == "revision_required":
                return WorkflowPhase.REVISION
            else:
                return WorkflowPhase.REJECTED
        
        # 默认：按顺序流转
        phase_order = [
            WorkflowPhase.IDLE,
            WorkflowPhase.MULTI_WRITING,
            WorkflowPhase.DRAFT_SELECTION,
            WorkflowPhase.DEEP_EDITING,
            WorkflowPhase.EXPERT_REVIEW,
            WorkflowPhase.PEER_REVIEW,
            WorkflowPhase.REVISION,
            WorkflowPhase.FINAL_DECISION,
            WorkflowPhase.PUBLISHED
        ]
        
        current_idx = phase_order.index(current)
        if current_idx < len(phase_order) - 1:
            return phase_order[current_idx + 1]
        
        return None
    
    async def _advance_phase(self, state: WorkflowState, next_phase: WorkflowPhase):
        """推进到下一阶段"""
        self.logger.info(f"[{state.workflow_id}] Advancing from {state.current_phase.value} to {next_phase.value}")
        state.current_phase = next_phase
        await self.coordinator.notify_phase_change(state.workflow_id, next_phase)
    
    async def _handle_phase_failure(
        self, 
        state: WorkflowState, 
        phase_result: PhaseResult
    ) -> bool:
        """
        处理阶段失败
        返回：是否成功恢复（可以继续）
        """
        # 实现重试逻辑
        failure_count = state.get_data("failure_count", 0) + 1
        state.set_data("failure_count", failure_count)
        
        max_retries = self.config.get("max_retries", 2)
        
        if failure_count <= max_retries:
            self.logger.warning(f"Phase {phase_result.phase.value} failed, retrying ({failure_count}/{max_retries})")
            await asyncio.sleep(2 ** failure_count)  # 指数退避
            return True
        else:
            self.logger.error(f"Phase {phase_result.phase.value} failed after {max_retries} retries")
            return False
    
    async def _check_timeout(self, state: WorkflowState) -> bool:
        """检查工作流是否超时"""
        total_timeout = self.config.get("total_timeout", 7200)  # 2小时
        elapsed = (datetime.now() - state.created_at).total_seconds()
        return elapsed > total_timeout
    
    def _calculate_consensus(self, verdicts: List[str]) -> Dict[str, Any]:
        """计算审稿共识"""
        if not verdicts:
            return {"level": 0, "dominant_verdict": "major_revision"}
        
        # 统计各裁决的数量
        counts = {}
        for v in verdicts:
            counts[v] = counts.get(v, 0) + 1
        
        # 找出主导裁决
        dominant = max(counts, key=counts.get)
        consensus_level = counts[dominant] / len(verdicts)
        
        return {
            "level": consensus_level,
            "dominant_verdict": dominant,
            "distribution": counts
        }
    
    def _generate_final_result(self, state: WorkflowState) -> Dict[str, Any]:
        """生成最终结果"""
        return {
            "workflow_id": state.workflow_id,
            "status": state.status.value,
            "final_phase": state.current_phase.value,
            "final_draft": state.get_data("final_manuscript") or state.get_data("current_draft"),
            "all_drafts": state.get_data("drafts", []),
            "review_history": state.get_data("review_reports", []),
            "revision_history": state.get_data("revision_history", []),
            "statistics": state.to_summary_dict(),
            "completion_time": (datetime.now() - state.created_at).total_seconds()
        }
```

## 2. 协调器 (Coordinator)

```python
class Coordinator:
    """
    协调器 - Agent集群的中央神经系统
    负责任务调度、状态管理、通信协调、冲突仲裁
    
    采用单例模式确保全局唯一
    """
    
    _instance: Optional['Coordinator'] = None
    _lock = asyncio.Lock()
    
    @classmethod
    async def get_instance(cls) -> 'Coordinator':
        """获取Coordinator单例"""
        if cls._instance is None:
            async with cls._lock:
                if cls._instance is None:
                    cls._instance = cls()
        return cls._instance
    
    def __init__(self):
        if Coordinator._instance is not None:
            raise RuntimeError("Use get_instance() to get Coordinator")
        
        self.logger = logging.getLogger(__name__)
        
        # Agent注册表
        self.agents: Dict[str, BaseAgent] = {}
        self.agent_status: Dict[str, AgentStatus] = {}
        
        # 工作流注册表
        self.workflows: Dict[str, WorkflowState] = {}
        
        # 消息总线
        self.message_bus: asyncio.Queue = asyncio.Queue()
        self.subscribers: Dict[str, List[str]] = {}  # message_type -> [agent_ids]
        
        # 事件处理器
        self.event_handlers: Dict[str, List[Callable]] = {}
        
        # 配置
        self.config: Dict[str, Any] = {}
        
        # 运行状态
        self.running = False
        self._bus_task: Optional[asyncio.Task] = None
    
    async def initialize(self, config: Dict[str, Any]):
        """初始化Coordinator"""
        self.config = config
        
        # 初始化Agent池
        await self._initialize_agent_pool(config)
        
        # 启动消息总线
        self.running = True
        self._bus_task = asyncio.create_task(self._message_bus_loop())
        
        self.logger.info("Coordinator initialized")
    
    async def _initialize_agent_pool(self, config: Dict[str, Any]):
        """初始化Agent池"""
        # 创建写作Agent
        writer_config = config.get("writer", {})
        for i in range(writer_config.get("count", 5)):
            agent_config = AgentConfig(
                agent_id=f"writer_{i}",
                role=AgentRole.WRITER,
                specialty=[writer_config.get("styles", ["balanced"])[i % 5]]
            )
            agent = WriterAgent(agent_config)
            await self.register_agent(agent)
        
        # 创建编辑Agent
        editor_config = config.get("editor", {})
        if editor_config.get("enabled", True):
            agent_config = AgentConfig(
                agent_id="editor_0",
                role=AgentRole.EDITOR,
                specialty=[editor_config.get("style_guide", "academic")]
            )
            agent = EditorAgent(agent_config)
            await self.register_agent(agent)
        
        # 创建专家Agent
        agent_config = AgentConfig(
            agent_id="expert_0",
            role=AgentRole.EXPERT,
            specialty=["domain_expert"]
        )
        agent = ExpertAgent(agent_config)
        await self.register_agent(agent)
        
        # 创建审稿人Agent
        reviewer_config = config.get("reviewer", {})
        for i in range(reviewer_config.get("count", 3)):
            agent_config = AgentConfig(
                agent_id=f"reviewer_{i}",
                role=AgentRole.REVIEWER,
                specialty=[f"reviewer_{i}"]
            )
            agent = ReviewerAgent(agent_config)
            await self.register_agent(agent)
        
        # 创建主编Agent
        agent_config = AgentConfig(
            agent_id="editor_in_chief_0",
            role=AgentRole.EDITOR_IN_CHIEF
        )
        agent = EditorInChiefAgent(agent_config)
        await self.register_agent(agent)
    
    async def register_agent(self, agent: BaseAgent) -> bool:
        """注册Agent到Coordinator"""
        agent_id = agent.config.agent_id
        
        if agent_id in self.agents:
            self.logger.warning(f"Agent {agent_id} already registered, updating")
        
        self.agents[agent_id] = agent
        self.agent_status[agent_id] = AgentStatus.IDLE
        
        self.logger.info(f"Agent {agent_id} registered")
        return True
    
    async def dispatch_task(
        self, 
        agent_role: str, 
        task: Dict[str, Any],
        timeout: Optional[int] = None
    ) -> Dict[str, Any]:
        """
        调度任务到指定类型的Agent
        
        Args:
            agent_role: Agent角色（writer/editor/expert/reviewer/editor_in_chief）
            task: 任务数据
            timeout: 超时时间（秒）
            
        Returns:
            任务执行结果
        """
        # 找到可用的Agent
        available_agents = [
            agent_id for agent_id, status in self.agent_status.items()
            if status == AgentStatus.IDLE and agent_id.startswith(agent_role)
        ]
        
        if not available_agents:
            raise RuntimeError(f"No available agent for role {agent_role}")
        
        # 选择第一个可用Agent（可扩展为负载均衡策略）
        selected_agent_id = available_agents[0]
        agent = self.agents[selected_agent_id]
        
        # 更新状态
        self.agent_status[selected_agent_id] = AgentStatus.BUSY
        
        try:
            # 执行任务
            result = await asyncio.wait_for(
                agent.process(task),
                timeout=timeout or agent.config.timeout
            )
            
            return result
            
        except asyncio.TimeoutError:
            raise RuntimeError(f"Task timeout for agent {selected_agent_id}")
            
        except Exception as e:
            self.agent_status[selected_agent_id] = AgentStatus.ERROR
            raise
            
        finally:
            # 恢复状态
            self.agent_status[selected_agent_id] = AgentStatus.IDLE
    
    async def route_message(self, message: AgentMessage) -> bool:
        """
        路由消息到目标Agent
        
        Args:
            message: 要路由的消息
            
        Returns:
            路由是否成功
        """
        receiver_id = message.header.receiver_id
        
        if receiver_id == "broadcast":
            # 广播消息
            await self._broadcast_message(message)
            return True
        
        if receiver_id not in self.agents:
            self.logger.error(f"Target agent {receiver_id} not found")
            return False
        
        target_agent = self.agents[receiver_id]
        
        # 投递消息到目标Agent的消息队列
        await target_agent.handle_message(message)
        
        return True
    
    async def _broadcast_message(self, message: AgentMessage):
        """广播消息到所有订阅者"""
        message_type = message.header.message_type.value
        subscribers = self.subscribers.get(message_type, [])
        
        for agent_id in subscribers:
            if agent_id in self.agents:
                # 创建消息副本
                msg_copy = AgentMessage.create(
                    sender_id=message.header.sender_id,
                    receiver_id=agent_id,
                    message_type=message.header.message_type,
                    payload=message.payload.copy(),
                    correlation_id=message.header.correlation_id
                )
                await self.agents[agent_id].handle_message(msg_copy)
    
    async def _message_bus_loop(self):
        """消息总线主循环"""
        while self.running:
            try:
                message = await asyncio.wait_for(
                    self.message_bus.get(),
                    timeout=1.0
                )
                
                # 处理消息
                await self.route_message(message)
                
            except asyncio.TimeoutError:
                continue
                
            except Exception as e:
                self.logger.exception("Message bus error")
    
    async def register_workflow(self, state: WorkflowState):
        """注册工作流到Coordinator"""
        self.workflows[state.workflow_id] = state
        self.logger.info(f"Workflow {state.workflow_id} registered")
    
    async def update_agent_status(self, agent_id: str, status: AgentStatus):
        """更新Agent状态"""
        self.agent_status[agent_id] = status
        
        # 发布状态变更事件
        await self._publish_event("agent_status_change", {
            "agent_id": agent_id,
            "status": status.value,
            "timestamp": datetime.now().isoformat()
        })
    
    async def notify_phase_change(self, workflow_id: str, new_phase: WorkflowPhase):
        """通知阶段变更"""
        await self._publish_event("phase_change", {
            "workflow_id": workflow_id,
            "new_phase": new_phase.value,
            "timestamp": datetime.now().isoformat()
        })
    
    async def _publish_event(self, event_type: str, data: Dict[str, Any]):
        """发布事件"""
        handlers = self.event_handlers.get(event_type, [])
        for handler in handlers:
            try:
                await handler(data)
            except Exception as e:
                self.logger.exception(f"Event handler error for {event_type}")
    
    def subscribe_to_event(self, event_type: str, handler: Callable):
        """订阅事件"""
        if event_type not in self.event_handlers:
            self.event_handlers[event_type] = []
        self.event_handlers[event_type].append(handler)
    
    async def shutdown(self):
        """关闭Coordinator"""
        self.running = False
        
        if self._bus_task:
            self._bus_task.cancel()
            try:
                await self._bus_task
            except asyncio.CancelledError:
                pass
        
        self.logger.info("Coordinator shutdown")
```

## 3. 冲突解决器 (ConflictResolver)

```python
class ConflictResolver:
    """
    冲突解决器
    处理多Agent协作中的冲突：
    - 评论冲突（审稿人给出矛盾意见）
    - 编辑冲突（不同编辑给出不同修改建议）
    - 引用冲突（对同一工作的引用冲突）
    - 术语冲突（术语使用不一致）
    """
    
    def __init__(self, coordinator: Coordinator):
        self.coordinator = coordinator
        self.logger = logging.getLogger(__name__)
    
    async def resolve_comment_conflicts(
        self, 
        comments: List[ReviewComment]
    ) -> List[ReviewComment]:
        """
        解决评论冲突
        
        策略：
        1. 识别矛盾评论
        2. 基于审稿人权重仲裁
        3. 无法仲裁时提交主编裁决
        """
        # 按位置分组
        location_groups: Dict[str, List[ReviewComment]] = {}
        for comment in comments:
            loc_key = self._location_to_key(comment.location)
            if loc_key not in location_groups:
                location_groups[loc_key] = []
            location_groups[loc_key].append(comment)
        
        resolved_comments = []
        conflicts_for_arbitration = []
        
        for loc_key, group in location_groups.items():
            if len(group) == 1:
                # 无冲突
                resolved_comments.extend(group)
            else:
                # 检查是否存在矛盾
                contradictions = self._find_contradictions(group)
                
                if not contradictions:
                    # 无矛盾，合并评论
                    merged = self._merge_comments(group)
                    resolved_comments.append(merged)
                else:
                    # 有矛盾，尝试仲裁
                    arbitration_result = await self._arbitrate(group, contradictions)
                    
                    if arbitration_result:
                        resolved_comments.append(arbitration_result)
                    else:
                        # 需要主编裁决
                        conflicts_for_arbitration.extend(group)
        
        # 提交无法解决的冲突给主编
        if conflicts_for_arbitration:
            chief_decision = await self._request_editor_in_chief_decision(
                conflicts_for_arbitration
            )
            resolved_comments.extend(chief_decision)
        
        return resolved_comments
    
    def _location_to_key(self, location: CommentLocation) -> str:
        """将位置转换为哈希键"""
        return f"{location.section_id}:{location.paragraph_index}"
    
    def _find_contradictions(
        self, 
        comments: List[ReviewComment]
    ) -> List[Tuple[ReviewComment, ReviewComment]]:
        """找出相互矛盾的评论对"""
        contradictions = []
        
        for i, c1 in enumerate(comments):
            for c2 in comments[i+1:]:
                # 检查是否矛盾
                if self._are_contradictory(c1, c2):
                    contradictions.append((c1, c2))
        
        return contradictions
    
    def _are_contradictory(self, c1: ReviewComment, c2: ReviewComment) -> bool:
        """判断两个评论是否矛盾"""
        # 简单启发式：相同位置，相反建议
        if c1.category == c2.category:
            # 例如：一个说"需要更多细节"，一个说"太冗长"
            opposing_pairs = [
                ("需要更多细节", "过于冗长"),
                ("缺少证明", "证明太长"),
                ("应该引用A", "不应该引用A")
            ]
            
            for p1, p2 in opposing_pairs:
                if (p1 in c1.content and p2 in c2.content) or \
                   (p2 in c1.content and p1 in c2.content):
                    return True
        
        return False
    
    def _merge_comments(self, comments: List[ReviewComment]) -> ReviewComment:
        """合并多个不冲突的评论"""
        # 选择最严重的评论作为主评论
        severity_order = [
            CommentSeverity.CRITICAL,
            CommentSeverity.MAJOR,
            CommentSeverity.MINOR,
            CommentSeverity.SUGGESTION
        ]
        
        primary = min(comments, key=lambda c: severity_order.index(c.severity))
        
        # 合并内容
        merged_content = primary.content + "\n\n" + \
            "\n".join([f"补充意见：{c.content}" for c in comments if c != primary])
        
        return ReviewComment(
            comment_id=str(uuid.uuid4()),
            reviewer_id="merged",
            draft_id=primary.draft_id,
            category=primary.category,
            severity=primary.severity,
            title=f"[合并] {primary.title}",
            content=merged_content,
            location=primary.location
        )
    
    async def _arbitrate(
        self, 
        comments: List[ReviewComment],
        contradictions: List[Tuple[ReviewComment, ReviewComment]]
    ) -> Optional[ReviewComment]:
        """
        基于权重仲裁
        
        仲裁策略：
        1. 计算每个评论的权重分数
        2. 选择权重最高的建议
        """
        # 简单权重：审稿人权威度 × 严重程度权重
        severity_weights = {
            CommentSeverity.CRITICAL: 10,
            CommentSeverity.MAJOR: 5,
            CommentSeverity.MINOR: 2,
            CommentSeverity.SUGGESTION: 1
        }
        
        scored_comments = []
        for comment in comments:
            weight = severity_weights.get(comment.severity, 1)
            scored_comments.append((weight, comment))
        
        # 按权重排序
        scored_comments.sort(reverse=True, key=lambda x: x[0])
        
        if len(scored_comments) >= 2:
            top_score, top_comment = scored_comments[0]
            second_score, _ = scored_comments[1]
            
            # 如果最高权重显著高于第二，选择最高
            if top_score > second_score * 1.5:
                return top_comment
        
        # 权重相近，无法仲裁
        return None
    
    async def _request_editor_in_chief_decision(
        self, 
        conflicts: List[ReviewComment]
    ) -> List[ReviewComment]:
        """请求主编裁决"""
        # 调度主编Agent进行裁决
        task = {
            "conflicts": conflicts,
            "task_type": "arbitration"
        }
        
        result = await self.coordinator.dispatch_task("editor_in_chief", task)
        
        # 解析裁决结果
        resolved = result.get("resolved_comments", [])
        
        return resolved
    
    async def resolve_edit_conflicts(
        self,
        edit_suggestions: List[Dict[str, Any]]
    ) -> List[Dict[str, Any]]:
        """
        解决编辑冲突
        多个编辑对同一段文本给出不同修改建议
        """
        # 按位置分组
        grouped = {}
        for suggestion in edit_suggestions:
            loc = suggestion.get("location")
            if loc not in grouped:
                grouped[loc] = []
            grouped[loc].append(suggestion)
        
        resolved = []
        for loc, group in grouped.items():
            if len(group) == 1:
                resolved.extend(group)
            else:
                # 冲突：选择最保守的修改（改动最小的）
                conservative = min(group, key=lambda s: len(s.get("suggested_change", "")))
                resolved.append(conservative)
        
        return resolved
```

```python
class HealthMonitor:
    """
    健康监控器
    监控系统健康状态，检测故障，触发恢复
    """
    
    def __init__(self, coordinator: Coordinator):
        self.coordinator = coordinator
        self.logger = logging.getLogger(__name__)
        self.check_interval = 30  # 秒
        self._monitor_task: Optional[asyncio.Task] = None
        self.running = False
    
    async def start(self):
        """启动监控"""
        self.running = True
        self._monitor_task = asyncio.create_task(self._monitor_loop())
        self.logger.info("Health monitor started")
    
    async def stop(self):
        """停止监控"""
        self.running = False
        if self._monitor_task:
            self._monitor_task.cancel()
            try:
                await self._monitor_task
            except asyncio.CancelledError:
                pass
        self.logger.info("Health monitor stopped")
    
    async def _monitor_loop(self):
        """监控主循环"""
        while self.running:
            try:
                await self._check_agents()
                await self._check_workflows()
                await asyncio.sleep(self.check_interval)
            except Exception as e:
                self.logger.exception("Health check error")
    
    async def _check_agents(self):
        """检查Agent健康状态"""
        for agent_id, agent in self.coordinator.agents.items():
            status = self.coordinator.agent_status.get(agent_id)
            
            if status == AgentStatus.ERROR:
                self.logger.warning(f"Agent {agent_id} in error state, attempting recovery")
                await self._recover_agent(agent_id)
            
            elif status == AgentStatus.BUSY:
                # 检查是否卡住
                # 实际实现：检查Agent的最后活动时间
                pass
    
    async def _check_workflows(self):
        """检查工作流健康"""
        for workflow_id, state in self.coordinator.workflows.items():
            # 检查是否超时
            phase_start = state.get_data("phase_start_time")
            if phase_start:
                elapsed = (datetime.now() - phase_start).total_seconds()
                timeout = state.config.get("phase_timeout", {}).get(
                    state.current_phase.value, 1800
                )
                
                if elapsed > timeout:
                    self.logger.warning(
                        f"Workflow {workflow_id} phase {state.current_phase.value} timeout"
                    )
                    await self._handle_workflow_timeout(workflow_id)
    
    async def _recover_agent(self, agent_id: str):
        """恢复故障Agent"""
        # 简单策略：重置Agent状态
        self.coordinator.agent_status[agent_id] = AgentStatus.IDLE
        self.logger.info(f"Agent {agent_id} recovered")
    
    async def _handle_workflow_timeout(self, workflow_id: str):
        """处理工作流超时"""
        state = self.coordinator.workflows.get(workflow_id)
        if state:
            state.status = WorkflowStatus.TIMEOUT
            # 通知相关人员
            await self.coordinator._publish_event("workflow_timeout", {
                "workflow_id": workflow_id,
                "phase": state.current_phase.value
            })
```
