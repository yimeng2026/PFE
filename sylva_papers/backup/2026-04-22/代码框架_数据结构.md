# Agent集群写稿系统 - 数据结构定义

## 1. 核心数据结构

### 1.1 文章草稿 (ArticleDraft)

```python
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Any
from datetime import datetime
from enum import Enum

class DraftStatus(Enum):
    RAW = "raw"                    # 原始初稿
    EDITED = "edited"              # 编辑后
    EXPERT_REVIEWED = "expert_reviewed"  # 专家审核后
    PEER_REVIEWED = "peer_reviewed"      # 同行审稿后
    REVISED = "revised"            # 修改后
    ACCEPTED = "accepted"          # 接受
    REJECTED = "rejected"          # 拒稿

class ContentFormat(Enum):
    MARKDOWN = "markdown"
    LATEX = "latex"
    PLAIN_TEXT = "plain_text"
    HTML = "html"

@dataclass
class Section:
    """论文章节"""
    section_id: str
    title: str
    section_type: str  # introduction, related_work, methodology, etc.
    content: str
    word_count: int
    order: int
    subsections: List['Section'] = field(default_factory=list)

@dataclass
class ArticleDraft:
    """文章草稿 - 核心数据实体"""
    
    # 标识信息
    draft_id: str
    version: int = 1
    parent_draft_id: Optional[str] = None  # 用于追踪修改历史
    
    # 元数据
    title: str = ""
    abstract: str = ""
    keywords: List[str] = field(default_factory=list)
    
    # 内容
    sections: List[Section] = field(default_factory=list)
    content: str = ""  # 完整内容（用于快速访问）
    format: ContentFormat = ContentFormat.MARKDOWN
    
    # 状态
    status: DraftStatus = DraftStatus.RAW
    
    # 质量评估
    quality_scores: Dict[str, float] = field(default_factory=dict)
    # 例如: {"novelty": 85, "rigor": 90, "clarity": 75, "completeness": 80}
    
    # 来源信息
    created_by: str = ""  # Agent ID
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    
    # 关联数据
    references: List[Dict] = field(default_factory=list)
    figures: List[Dict] = field(default_factory=list)
    tables: List[Dict] = field(default_factory=list)
    
    # 元信息（用于追踪和调试）
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    def get_word_count(self) -> int:
        """计算总字数"""
        return sum(s.word_count for s in self.sections)
    
    def get_section_by_type(self, section_type: str) -> Optional[Section]:
        """按类型获取章节"""
        for section in self.sections:
            if section.section_type == section_type:
                return section
        return None
    
    def calculate_overall_score(self) -> float:
        """计算综合质量分"""
        if not self.quality_scores:
            return 0.0
        return sum(self.quality_scores.values()) / len(self.quality_scores)
    
    def to_dict(self) -> Dict:
        """序列化为字典"""
        return {
            "draft_id": self.draft_id,
            "version": self.version,
            "title": self.title,
            "abstract": self.abstract,
            "keywords": self.keywords,
            "status": self.status.value,
            "quality_scores": self.quality_scores,
            "created_by": self.created_by,
            "created_at": self.created_at.isoformat(),
            "word_count": self.get_word_count(),
            "overall_score": self.calculate_overall_score()
        }
```

### 1.2 评审意见 (ReviewComment)

```python
class CommentSeverity(Enum):
    CRITICAL = "critical"      # 致命问题，必须解决
    MAJOR = "major"            # 重要问题
    MINOR = "minor"            # 小问题
    SUGGESTION = "suggestion"  # 建议性意见
    PRAISE = "praise"          # 表扬/肯定

class CommentCategory(Enum):
    TECHNICAL = "technical"        # 技术问题
    METHODOLOGY = "methodology"    # 方法论
    WRITING = "writing"            # 写作/表达
    STRUCTURE = "structure"        # 结构
    NOVELTY = "novelty"            # 创新性
    RELATED_WORK = "related_work"  # 相关工作
    EXPERIMENT = "experiment"      # 实验/验证

@dataclass
class CommentLocation:
    """评论位置信息"""
    section_id: Optional[str] = None
    paragraph_index: Optional[int] = None
    line_start: Optional[int] = None
    line_end: Optional[int] = None
    quote_text: Optional[str] = None  # 引用原文

@dataclass
class ReviewComment:
    """评审意见/评论"""
    
    comment_id: str
    reviewer_id: str
    draft_id: str
    
    # 评论内容
    category: CommentCategory
    severity: CommentSeverity
    title: str  # 简短标题
    content: str  # 详细内容
    
    # 位置信息
    location: CommentLocation = field(default_factory=CommentLocation)
    
    # 建议修改（可选）
    suggested_change: Optional[str] = None
    
    # 状态追踪
    status: str = "open"  # open, addressed, rejected, disputed
    response: Optional[str] = None  # 作者的回应
    
    # 元数据
    created_at: datetime = field(default_factory=datetime.now)
    resolved_at: Optional[datetime] = None
    
    # 关联其他评论（例如相关评论）
    related_comments: List[str] = field(default_factory=list)

@dataclass
class ReviewReport:
    """完整审稿报告"""
    
    report_id: str
    reviewer_id: str
    draft_id: str
    
    # 总体评价
    summary: str  # 论文概述
    overall_assessment: str  # 总体评估
    
    # 详细评价
    strengths: List[str] = field(default_factory=list)  # 优点
    weaknesses: List[str] = field(default_factory=list)  # 缺点
    
    # 具体评论列表
    comments: List[ReviewComment] = field(default_factory=list)
    
    # 分类统计
    comment_stats: Dict[str, int] = field(default_factory=dict)
    # 例如: {"critical": 2, "major": 5, "minor": 8}
    
    # 裁决建议
    recommendation: str = ""  # accept, minor_revision, major_revision, reject
    confidence_level: float = 0.8  # 置信度 0-1
    
    # 元数据
    review_type: str = "peer_review"  # peer_review, expert_review, editorial_review
    created_at: datetime = field(default_factory=datetime.now)
    
    def get_comments_by_severity(self, severity: CommentSeverity) -> List[ReviewComment]:
        """按严重程度筛选评论"""
        return [c for c in self.comments if c.severity == severity]
    
    def get_open_comments(self) -> List[ReviewComment]:
        """获取未解决的评论"""
        return [c for c in self.comments if c.status == "open"]
    
    def calculate_score(self) -> float:
        """基于评论计算分数（越低越好）"""
        weights = {
            CommentSeverity.CRITICAL: 10,
            CommentSeverity.MAJOR: 5,
            CommentSeverity.MINOR: 1,
            CommentSeverity.SUGGESTION: 0.5
        }
        total = sum(weights.get(c.severity, 1) for c in self.comments)
        return max(0, 100 - total)
```

### 1.3 工作流状态 (WorkflowState)

```python
class WorkflowPhase(Enum):
    IDLE = "idle"                          # 空闲
    MULTI_WRITING = "multi_writing"      # 多写并行
    DRAFT_SELECTION = "draft_selection"  # 初稿筛选
    DEEP_EDITING = "deep_editing"        # 深度编辑
    EXPERT_REVIEW = "expert_review"      # 专家评审
    PEER_REVIEW = "peer_review"          # 同行审稿
    REVISION = "revision"                # 修改阶段
    FINAL_DECISION = "final_decision"    # 最终决策
    PUBLISHED = "published"              # 发布/完成
    REJECTED = "rejected"                # 拒稿

class WorkflowStatus(Enum):
    RUNNING = "running"
    PAUSED = "paused"
    COMPLETED = "completed"
    FAILED = "failed"
    TIMEOUT = "timeout"

@dataclass
class PhaseResult:
    """阶段执行结果"""
    phase: WorkflowPhase
    status: str  # success, partial_success, failed
    output_data: Dict[str, Any] = field(default_factory=dict)
    error_message: Optional[str] = None
    duration_seconds: float = 0.0
    agent_contributions: Dict[str, Any] = field(default_factory=dict)

@dataclass
class WorkflowState:
    """工作流状态 - 整个系统的中央状态"""
    
    workflow_id: str
    topic: str  # 论文主题
    
    # 当前状态
    current_phase: WorkflowPhase = WorkflowPhase.IDLE
    status: WorkflowStatus = WorkflowStatus.RUNNING
    
    # 历史记录
    phase_history: List[PhaseResult] = field(default_factory=list)
    
    # 当前阶段数据
    current_data: Dict[str, Any] = field(default_factory=dict)
    
    # 全局数据存储
    data_store: Dict[str, Any] = field(default_factory=dict)
    # 键值存储，例如:
    # "drafts": List[ArticleDraft] - 所有初稿
    # "selected_draft": ArticleDraft - 选中初稿
    # "review_reports": List[ReviewReport] - 审稿报告
    
    # 活跃Agent
    active_agents: List[str] = field(default_factory=list)
    agent_assignments: Dict[str, str] = field(default_factory=dict)  # agent_id -> task_id
    
    # 统计信息
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    estimated_completion: Optional[datetime] = None
    
    # 配置
    config: Dict[str, Any] = field(default_factory=dict)
    
    # 元数据
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    def transition_to(self, new_phase: WorkflowPhase, result: Optional[PhaseResult] = None):
        """状态转换"""
        if result:
            self.phase_history.append(result)
        self.current_phase = new_phase
        self.updated_at = datetime.now()
    
    def get_data(self, key: str, default=None) -> Any:
        """获取数据"""
        return self.data_store.get(key, default)
    
    def set_data(self, key: str, value: Any):
        """设置数据"""
        self.data_store[key] = value
        self.updated_at = datetime.now()
    
    def get_current_draft(self) -> Optional[ArticleDraft]:
        """获取当前主草稿"""
        return self.data_store.get("current_draft")
    
    def get_all_drafts(self) -> List[ArticleDraft]:
        """获取所有草稿"""
        return self.data_store.get("drafts", [])
    
    def get_review_reports(self) -> List[ReviewReport]:
        """获取所有审稿报告"""
        return self.data_store.get("review_reports", [])
    
    def to_summary_dict(self) -> Dict:
        """生成状态摘要"""
        return {
            "workflow_id": self.workflow_id,
            "current_phase": self.current_phase.value,
            "status": self.status.value,
            "draft_count": len(self.get_all_drafts()),
            "review_count": len(self.get_review_reports()),
            "active_agents": len(self.active_agents),
            "elapsed_time": (datetime.now() - self.created_at).total_seconds(),
            "progress_percentage": self._calculate_progress()
        }
    
    def _calculate_progress(self) -> int:
        """计算进度百分比"""
        phase_order = [
            WorkflowPhase.IDLE,
            WorkflowPhase.MULTI_WRITING,
            WorkflowPhase.DRAFT_SELECTION,
            WorkflowPhase.DEEP_EDITING,
            WorkflowPhase.EXPERT_REVIEW,
            WorkflowPhase.PEER_REVIEW,
            WorkflowPhase.REVISION,
            WorkflowPhase.FINAL_DECISION,
            WorkflowPhase.PUBLISHED
        ]
        if self.current_phase in phase_order:
            return int((phase_order.index(self.current_phase) / (len(phase_order) - 1)) * 100)
        return 0
```

### 1.4 Agent消息 (AgentMessage)

```python
class MessageType(Enum):
    # 任务相关
    TASK_ASSIGN = "task_assign"
    TASK_COMPLETE = "task_complete"
    TASK_FAILED = "task_failed"
    TASK_CANCEL = "task_cancel"
    
    # 写作相关
    SUBMIT_DRAFT = "submit_draft"
    DRAFT_ACCEPTED = "draft_accepted"
    DRAFT_REJECTED = "draft_rejected"
    
    # 审稿相关
    REQUEST_REVIEW = "request_review"
    SUBMIT_REVIEW = "submit_review"
    REVIEW_CONFLICT = "review_conflict"
    
    # 修改相关
    REQUEST_REVISION = "request_revision"
    SUBMIT_REVISION = "submit_revision"
    
    # 决策相关
    REQUEST_DECISION = "request_decision"
    FINAL_DECISION = "final_decision"
    
    # 协调相关
    AGENT_REGISTER = "agent_register"
    AGENT_STATUS = "agent_status"
    HEARTBEAT = "heartbeat"
    
    # 系统相关
    SYSTEM_ERROR = "system_error"
    TIMEOUT_WARNING = "timeout_warning"
    PROGRESS_UPDATE = "progress_update"

@dataclass
class MessageHeader:
    """消息头"""
    message_id: str
    correlation_id: Optional[str]  # 用于关联请求-响应
    sender_id: str
    receiver_id: str  # 或 "broadcast" 表示广播
    message_type: MessageType
    timestamp: datetime
    priority: int  # 1-10
    ttl: int = 3  # 消息生存时间（重试次数）

@dataclass
class AgentMessage:
    """Agent间通信消息 - 完整定义"""
    
    header: MessageHeader
    payload: Dict[str, Any] = field(default_factory=dict)
    
    # 追踪信息
    delivery_attempts: int = 0
    delivered_at: Optional[datetime] = None
    
    @classmethod
    def create(
        cls,
        sender_id: str,
        receiver_id: str,
        message_type: MessageType,
        payload: Dict[str, Any],
        priority: int = 5,
        correlation_id: Optional[str] = None
    ) -> "AgentMessage":
        """工厂方法创建消息"""
        header = MessageHeader(
            message_id=str(uuid.uuid4()),
            correlation_id=correlation_id,
            sender_id=sender_id,
            receiver_id=receiver_id,
            message_type=message_type,
            timestamp=datetime.now(),
            priority=priority
        )
        return cls(header=header, payload=payload)
    
    def create_response(
        self,
        message_type: MessageType,
        payload: Dict[str, Any]
    ) -> "AgentMessage":
        """创建响应消息（自动关联）"""
        return AgentMessage.create(
            sender_id=self.header.receiver_id,
            receiver_id=self.header.sender_id,
            message_type=message_type,
            payload=payload,
            correlation_id=self.header.message_id
        )
    
    def is_broadcast(self) -> bool:
        """是否为广播消息"""
        return self.header.receiver_id == "broadcast"
    
    def to_dict(self) -> Dict:
        """序列化"""
        return {
            "header": {
                "message_id": self.header.message_id,
                "correlation_id": self.header.correlation_id,
                "sender_id": self.header.sender_id,
                "receiver_id": self.header.receiver_id,
                "message_type": self.header.message_type.value,
                "timestamp": self.header.timestamp.isoformat(),
                "priority": self.header.priority,
                "ttl": self.header.ttl
            },
            "payload": self.payload,
            "delivery_attempts": self.delivery_attempts
        }
```

## 2. 质量评估体系

@dataclass
class QualityDimension:
    """质量维度定义"""
    name: str
    description: str
    weight: float  # 权重 0-1
    scoring_criteria: Dict[int, str]  # 分数 -> 描述

@dataclass
class QualityAssessment:
    """质量评估结果"""
    
    assessment_id: str
    target_id: str  # 被评估对象ID
    assessor_id: str  # 评估者ID
    
    # 多维度评分
    dimension_scores: Dict[str, float] = field(default_factory=dict)
    # 例如: {"novelty": 85, "rigor": 90, "clarity": 75}
    
    # 加权总分
    overall_score: float = 0.0
    
    # 详细反馈
    detailed_feedback: Dict[str, str] = field(default_factory=dict)
    
    # 改进建议
    improvement_suggestions: List[str] = field(default_factory=list)
    
    # 评估时间
    assessed_at: datetime = field(default_factory=datetime.now)
    
    def calculate_weighted_score(self, weights: Dict[str, float]) -> float:
        """基于权重计算加权分"""
        total = 0.0
        weight_sum = 0.0
        
        for dim, score in self.dimension_scores.items():
            w = weights.get(dim, 1.0)
            total += score * w
            weight_sum += w
        
        return total / weight_sum if weight_sum > 0 else 0.0

## 3. 配置数据结构

```python
@dataclass
class WriterConfig:
    """写作Agent配置"""
    count: int = 5  # 并行写作Agent数量
    styles: List[str] = field(default_factory=lambda: [
        "theoretical", "algorithmic", "applied", "historical", "innovative"
    ])
    models: List[str] = field(default_factory=lambda: ["default"])
    timeout_seconds: int = 600

@dataclass
class EditorConfig:
    """编辑Agent配置"""
    enabled: bool = True
    style_guide: str = "academic"
    auto_apply_edits: bool = False  # 是否自动应用编辑建议
    quality_threshold: int = 70

@dataclass
class ReviewerConfig:
    """审稿人Agent配置"""
    count: int = 3  # 并行审稿人数量
    review_styles: List[str] = field(default_factory=lambda: [
        "theoretical", "practical", "gatekeeper", "interdisciplinary"
    ])
    consensus_threshold: float = 0.6  # 共识阈值

@dataclass
class WorkflowConfig:
    """工作流配置"""
    max_iterations: int = 3  # 最大修改轮数
    phase_timeout: Dict[str, int] = field(default_factory=lambda: {
        "multi_writing": 1800,
        "deep_editing": 900,
        "expert_review": 1200,
        "peer_review": 1800,
        "revision": 1800
    })
    auto_advance: bool = True  # 是否自动进入下一阶段

@dataclass
class QualityThresholds:
    """质量阈值配置"""
    accept_threshold: int = 80  # 直接接受阈值
    minor_revision_threshold: int = 60  # 小修阈值
    major_revision_threshold: int = 40  # 大修阈值
    # <40 拒稿

@dataclass
class SystemConfig:
    """系统整体配置"""
    
    system_name: str = "AgentWritingSystem"
    version: str = "1.0.0"
    
    # 各组件配置
    writer: WriterConfig = field(default_factory=WriterConfig)
    editor: EditorConfig = field(default_factory=EditorConfig)
    reviewer: ReviewerConfig = field(default_factory=ReviewerConfig)
    workflow: WorkflowConfig = field(default_factory=WorkflowConfig)
    quality: QualityThresholds = field(default_factory=QualityThresholds)
    
    # 输出配置
    output_format: str = "markdown"  # markdown, latex
    output_directory: str = "./outputs"
    save_intermediate: bool = True  # 是否保存中间结果
    
    # 日志配置
    log_level: str = "INFO"
    log_file: Optional[str] = None
    
    def to_yaml(self) -> str:
        """导出为YAML格式"""
        import yaml
        return yaml.dump(self.to_dict(), default_flow_style=False)
    
    def to_dict(self) -> Dict:
        """转换为字典"""
        from dataclasses import asdict
        return asdict(self)
    
    @classmethod
    def from_dict(cls, data: Dict) -> "SystemConfig":
        """从字典加载"""
        # 实现反序列化逻辑
        return cls(**data)
```
